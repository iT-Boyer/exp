\newcommand{\collatz}
{
 \FPclip \n \n
 \FPprint \n
 \FPifeq \n 1
 \else

    \FPmul \n 3 \n
     \FPadd \n \n 1
     \collatz 
 \else
    \FPdiv \n \n 2
     \collatz 
 \fi
 \fi
}
